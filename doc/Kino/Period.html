def overlap?(some)
  @timerange.cover?(some.timerange.to_a.first) && (cur_hall & some.cur_hall).any?
end

def to_s
  "Время: #{@timerange}
  Описание: #{@per_description}
  Кино: #{@filters.nil? ? @title : @filters}
  Цена: #{@price}
  Зал: #{@hall}"
end
